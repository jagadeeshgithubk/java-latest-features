public static String getCommaSeperatedSortedListAsString(String incCommunity) {
		incCommunity=incCommunity!=null?incCommunity:"";
		String commentPrefix ="";
		if(incCommunity.contains(":")) {
			int startingIndex = incCommunity.indexOf(":")+1;
			commentPrefix = incCommunity.substring(0,startingIndex).trim();
			String incCommunityBeforeSort = incCommunity.substring(startingIndex,incCommunity.length());
			incCommunity=incCommunityBeforeSort;
		}
		String[] communitiesAfterSplit = incCommunity.split(",");
		List<String> communitiesList = new ArrayList<String>();
		for (String value : communitiesAfterSplit) {
			communitiesList.add(value.trim());
		}
		Collections.sort(communitiesList,new Comparator<String>() {
			public int compare(String o1, String o2) {
				return o1.compareToIgnoreCase(o2);
			}
		});
		
		return (commentPrefix+" "+getListAsCommaAppendedString(communitiesList)).trim();
	}

         Bsc.Book_Code,
         bsc.status
     FROM
         book_scope_cmptape bsc 
     WHERE
         (
             BSC.BOOK_CODE, BSC.BOOK_CMPTAPE_ID,1
         ) IN (
             SELECT
                 BOOK_CODE,
                 BOOK_CMPTAPE_ID,
                 ROW_NUMBER() OVER (PARTITION 
             BY
                 BOOK_CODE 
             ORDER BY
                 PRIORITY) rn 
             from
                 (SELECT
                     BOOK_CODE,
                     BOOK_CMPTAPE_ID,
                     (CASE STATUS 
                         WHEN 'C' THEN 1 
                         WHEN 'E' THEN 2 
                         WHEN 'A' THEN 3  
                         WHEN 'W' THEN 4 
                         WHEN 'R' THEN 5 
                         ELSE 0 
                     END)PRIORITY 
                 FROM
                     BOOK_SCOPE_CMPTAPE 
                 WHERE
                     LIST_TYPE IS NOT NULL 
                     AND UPPER(STATUS)IN (
                         'A','E','R','C','W'
                     ))) 
             ORDER BY
                 BSC.BOOK_CODE
